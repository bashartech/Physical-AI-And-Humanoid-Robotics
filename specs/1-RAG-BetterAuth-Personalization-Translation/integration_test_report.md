            # ... other profile fields
        })

        # Test personalization

        personalized = self.backend_service.personalize(
            content, user_data["token"]
        )

        # Test translation

            content, "ur", user_data["token"]
        )

        # Test chat

        )

        assert translated is not None
        assert response is not None
